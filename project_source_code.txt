import React, { useState, useEffect, useMemo } from "react";
import {
  FileText,
  Printer,
  Download,
  RefreshCw,
  CheckCircle2,
  AlertCircle,
  Trash2,
  Plus,
  ChevronLeft,
  ChevronRight,
  Sparkles,
  Layers,
  FileSpreadsheet,
  Info,
  Calendar,
  User,
  ShieldCheck,
  Check,
  RotateCcw
} from "lucide-react";
import { motion, AnimatePresence } from "motion/react";
import { FormPage1, MaterialItem, SecurityPermitFormData } from "./types";
import { convertNumberToArabicWords, getTodayDateString } from "./utils";
import IraqiCrest from "./components/IraqiCrest";

const INITIAL_PAGE1: FormPage1 = {
  authName: "",
  authId: "",
  driverName: "",
  driverId: "",
  vehicleType: "",
  vehicleNumber: "",
  materialSource: "",
  date: getTodayDateString(),
  materialCountNum: "",
  materialCountText: "",
  entryExit: "entry"
};

const INITIAL_ITEMS = (): MaterialItem[] => {
  return Array.from({ length: 25 }, (_, i) => ({
    id: i + 1,
    name: "",
    note: ""
  }));
};

export default function App() {
  const [activeTab, setActiveTab] = useState<"step1" | "step2" | "preview">("step1");
  const [formData, setFormData] = useState<SecurityPermitFormData>({
    page1: { ...INITIAL_PAGE1 },
    items: INITIAL_ITEMS()
  });

  const [checklist, setChecklist] = useState({
    officialLetter: false,
    idVerified: false,
    vehicleRegistered: false,
    timingCheck: false,
    weeklyHolidayCheck: true
  });

  const [notification, setNotification] = useState<{
    message: string;
    type: "success" | "info" | "error";
  } | null>(null);

  useEffect(() => {
    const saved = localStorage.getItem("ministry_permit_data");
    if (saved) {
      try {
        const parsed = JSON.parse(saved);
        if (parsed.page1 && Array.isArray(parsed.items)) {
          let loadedItems = [...parsed.items];
          if (loadedItems.length < 25) {
            const extra = Array.from({ length: 25 - loadedItems.length }, (_, i) => ({
              id: loadedItems.length + i + 1,
              name: "",
              note: ""
            }));
            loadedItems = [...loadedItems, ...extra];
          } else if (loadedItems.length > 25) {
            loadedItems = loadedItems.slice(0, 25);
          }
          setFormData({ page1: parsed.page1, items: loadedItems });
          showNotification("تم استعادة المسودة الأخيرة المحفوظة تلقائياً ✅", "info");
        }
      } catch (e) {
        console.error("Error loading localStorage data", e);
      }
    }
  }, []);

  const showNotification = (message: string, type: "success" | "info" | "error" = "success") => {
    setNotification({ message, type });
    setTimeout(() => {
      setNotification(null);
    }, 4000);
  };

  const handlePage1Change = (field: keyof FormPage1, value: string) => {
    setFormData((prev) => ({
      ...prev,
      page1: {
        ...prev.page1,
        [field]: value
      }
    }));
  };

  const handleItemChange = (index: number, field: "name" | "note", value: string) => {
    setFormData((prev) => {
      const updatedItems = [...prev.items];
      updatedItems[index] = {
        ...updatedItems[index],
        [field]: value
      };
      return {
        ...prev,
        items: updatedItems
      };
    });
  };

  const loadDemoData = () => {
    const demoItems = INITIAL_ITEMS();
    demoItems[0] = { id: 1, name: "شاشات عرض ذكية LED قياس 65 بوصة", note: "تجهيز قسم تكنولوجيا المعلومات لملعب المدينة الدولي" };
    demoItems[1] = { id: 2, name: "حواسيب محمولة HP ProBook", note: "عدد 5 حاسبة مخصصة لأكاديمية الشباب والرياضة" };
    demoItems[2] = { id: 3, name: "أطقم ملابس ومدربين رياضية رسمية مخصصة", note: "مخصصة للمنتخب الأولمبي للشباب العراقي" };
    demoItems[3] = { id: 4, name: "أدوات ومستلزمات بث مرئي وصوت مع الكابلات", note: "خاصة بقناة العراق الرياضية لتغطية المباراة الودية" };
    demoItems[4] = { id: 5, name: "كرات قدم معتمدة من فيفا (Pro)", note: "عدد 40 كرة مخصصة لملاعب التدريب الفرعية" };

    const filledDemo: SecurityPermitFormData = {
      page1: {
        authName: "أحمد جاسم عبد الرضا الجبوري",
        authId: "IQ-883920192-M",
        driverName: "مصطفى قاسم كريم الخفاجي",
        driverId: "IQ-774920230-T",
        vehicleType: "شاحنة حمل كيا متوسطة",
        vehicleNumber: "بغداد - 66324 / ب",
        materialSource: "مخازن الوزارة الكبرى - المدينة الشبابية",
        date: getTodayDateString(),
        materialCountNum: "5",
        materialCountText: "خمسة مواد فقط لا غير",
        entryExit: "exit"
      },
      items: demoItems
    };

    setFormData(filledDemo);
    setChecklist({
      officialLetter: true,
      idVerified: true,
      vehicleRegistered: true,
      timingCheck: true,
      weeklyHolidayCheck: true
    });

    showNotification("تم ملء الاستمارة ببيانات تجريبية رسمية نموذجية ✨", "success");
  };

  const saveDraft = () => {
    localStorage.setItem("ministry_permit_data", JSON.stringify(formData));
    showNotification("تم حفظ بيانات الاستمارتين في الذاكرة المحلية بنجاح 💾", "success");
  };

  const clearForm = () => {
    if (window.confirm("هل أنت متأكد من رغبتك في تفريغ كافة الحقول وإعادة تصفير الاستمارة؟")) {
      setFormData({
        page1: { ...INITIAL_PAGE1 },
        items: INITIAL_ITEMS()
      });
      setChecklist({
        officialLetter: false,
        idVerified: false,
        vehicleRegistered: false,
        timingCheck: false,
        weeklyHolidayCheck: true
      });
      localStorage.removeItem("ministry_permit_data");
      showNotification("تم تصفير الاستمارة ومسح المسودات 🗑️", "info");
    }
  };

  const filledItemsCount = useMemo(() => {
    return formData.items.filter((item) => item.name.trim() !== "").length;
  }, [formData.items]);

  const autoSyncMaterialCount = () => {
    const count = filledItemsCount;
    if (count === 0) {
      showNotification("الرجاء إدخال مادة واحدة على الأقل في الصفحة الثانية أولاً ⚠️", "error");
      return;
    }
    const countText = convertNumberToArabicWords(count);
    setFormData((prev) => ({
      ...prev,
      page1: {
        ...prev.page1,
        materialCountNum: String(count),
        materialCountText: countText
      }
    }));
    showNotification(`تمت مزامنة العدد: تلقائياً تم حساب (${count}) مواد وكتابتها بالحروف! 🖋️`, "success");
  };

  const triggerPrint = (isPDF: boolean = false) => {
    if (isPDF) {
      showNotification("جاري فتح واجهة تصدير PDF... يرجى اختيار 'حفظ بتنسيق PDF' كوجهة طباعة 📄", "success");
    } else {
      showNotification("جاري تشغيل واجهة الطباعة الفورية... 🖨️", "success");
    }
    
    if (filledItemsCount > 0 && !formData.page1.materialCountNum) {
      const count = filledItemsCount;
      const countText = convertNumberToArabicWords(count);
      setFormData((prev) => ({
        ...prev,
        page1: {
          ...prev.page1,
          materialCountNum: String(count),
          materialCountText: countText
        }
      }));
    }

    setTimeout(() => {
      window.print();
    }, 450);
  };

  const completeness = useMemo(() => {
    const fields = [
      formData.page1.authName,
      formData.page1.authId,
      formData.page1.driverName,
      formData.page1.driverId,
      formData.page1.vehicleType,
      formData.page1.vehicleNumber,
      formData.page1.materialSource,
      formData.page1.date,
      formData.page1.materialCountNum,
      formData.page1.materialCountText
    ];
    const filledFieldsCount = fields.filter((f) => f.trim() !== "").length;
    const itemsCount = filledItemsCount > 0 ? 1 : 0;
    const exitOptionFilled = formData.page1.entryExit !== "" ? 1 : 0;
    
    return Math.round(((filledFieldsCount + itemsCount + exitOptionFilled) / 12) * 100);
  }, [formData, filledItemsCount]);

  return (
    <div className="min-h-screen bg-slate-100 flex flex-col antialiased">
      {/* شريط الإدارة والتحكم الذكي - معطل أثناء الطباعة بنسبة 100% */}
      <header className="no-print bg-slate-900 text-white shadow-xl border-b border-amber-500 sticky top-0 z-40">
        <div className="max-w-7xl mx-auto px-4 py-3 sm:px-6 lg:px-8">
          <div className="flex flex-col md:flex-row md:items-center md:justify-between gap-4">
            
            <div className="flex items-center gap-3">
              <div className="bg-white p-1 rounded-full shadow-lg border border-amber-400">
                <IraqiCrest size={46} className="text-blue-900" />
              </div>
              <div>
                <div className="flex items-center gap-2">
                  <span className="bg-amber-500 text-xs font-bold text-slate-900 px-2 py-0.5 rounded-sm">
                    وزارة الشباب والرياضة
                  </span>
                  <span className="text-xs text-slate-400">مكتب التصاريح الأمنية</span>
                </div>
                <h1 className="text-lg font-bold text-white tracking-tight">
                  المنظومة الإلكترونية الذكية لإستمارات التصاريح الأمنية للمواد
                </h1>
              </div>
            </div>

            <div className="flex flex-wrap items-center gap-2">
              <button
                onClick={loadDemoData}
                className="flex items-center gap-1 text-xs bg-indigo-700 hover:bg-indigo-600 font-medium py-2 px-3 rounded-md transition-all text-white border border-indigo-500 cursor-pointer"
              >
                <Sparkles size={14} className="animate-pulse" />
                <span>ملء عينة نموذجية</span>
              </button>

              <button
                onClick={saveDraft}
                className="flex items-center gap-1 text-xs bg-emerald-700 hover:bg-emerald-600 font-medium py-2 px-3 rounded-md transition-all text-white border border-emerald-500 cursor-pointer"
              >
                <Download size={14} />
                <span>حفظ المسودة</span>
              </button>

              <button
                onClick={clearForm}
                className="flex items-center gap-1 text-xs bg-rose-700 hover:bg-rose-600 font-medium py-2 px-3 rounded-md transition-all text-white border border-rose-500 cursor-pointer"
              >
                <Trash2 size={14} />
                <span>تصفير الاستمارة</span>
              </button>

              <button
                onClick={() => triggerPrint(false)}
                className="flex items-center gap-1.5 text-xs bg-amber-500 hover:bg-amber-400 font-bold py-2 px-4 rounded-md transition-all text-slate-900 shadow-md cursor-pointer scale-105"
              >
                <Printer size={15} />
                <span>طباعة / تصدير PDF</span>
              </button>
            </div>

          </div>

          <div className="flex items-center justify-between border-t border-slate-800 mt-3 pt-2">
            <div className="flex gap-1.5 overflow-x-auto pb-1 max-w-full">
              <button
                onClick={() => setActiveTab("step1")}
                className={`flex items-center gap-2 py-2 px-4 rounded-md text-xs font-semibold transition-all cursor-pointer whitespace-nowrap ${
                  activeTab === "step1"
                    ? "bg-amber-500 text-slate-950 px-5 shadow"
                    : "bg-slate-800 hover:bg-slate-700 text-slate-300"
                }`}
              >
                <span className="flex items-center justify-center w-5 h-5 rounded-full bg-slate-950 text-amber-400 text-[10px] font-bold">١</span>
                الورقة الأولى: معلومات الموظف والعجلة
              </button>

              <button
                onClick={() => setActiveTab("step2")}
                className={`flex items-center gap-2 py-2 px-4 rounded-md text-xs font-semibold transition-all cursor-pointer whitespace-nowrap ${
                  activeTab === "step2"
                    ? "bg-amber-500 text-slate-950 px-5 shadow"
                    : "bg-slate-800 hover:bg-slate-700 text-slate-300"
                }`}
              >
                <span className="flex items-center justify-center w-5 h-5 rounded-full bg-slate-950 text-amber-400 text-[10px] font-bold">٢</span>
                الورقة الثانية: جدول المواد وقائمة الكميات 
                {filledItemsCount > 0 && <span className="bg-blue-600 text-white rounded-full text-[9px] px-1.5 mr-1">{filledItemsCount}</span>}
              </button>

              <button
                onClick={() => setActiveTab("preview")}
                className={`flex items-center gap-2 py-2 px-4 rounded-md text-xs font-semibold transition-all cursor-pointer whitespace-nowrap ${
                  activeTab === "preview"
                    ? "bg-amber-500 text-slate-950 px-5 shadow"
                    : "bg-slate-800 hover:bg-slate-700 text-slate-300"
                }`}
              >
                <Layers size={14} />
                معاينة الصفحتين جنباً إلى جنب (كـ A4 للطباعة)
              </button>
            </div>

            <div className="hidden sm:flex items-center gap-3 mr-4">
              <div className="text-left">
                <div className="text-[10px] text-slate-400">نسبة اكتمال الملء</div>
                <div className="text-xs font-mono font-bold text-amber-400">{completeness}%</div>
              </div>
              <div className="w-24 bg-slate-800 rounded-full h-2 overflow-hidden">
                <div
                  className="bg-amber-500 h-2 rounded-full transition-all duration-500"
                  style={{ width: `${completeness}%` }}
                ></div>
              </div>
            </div>

          </div>
        </div>
      </header>

      <AnimatePresence>
        {notification && (
          <motion.div
            initial={{ opacity: 0, y: -50 }}
            animate={{ opacity: 1, y: 0 }}
            exit={{ opacity: 0, y: -20 }}
            className="no-print fixed top-24 left-4 right-4 md:left-auto md:w-96 bg-slate-900 border-r-4 border-amber-500 text-white rounded-lg p-3 shadow-2xl flex items-center gap-3 z-50 pointer-events-auto"
          >
            <div className="text-amber-500 flex-shrink-0">
              <CheckCircle2 size={18} />
            </div>
            <p className="text-xs font-medium font-sans leading-relaxed">{notification.message}</p>
          </motion.div>
        )}
      </AnimatePresence>

      <main className="flex-1 max-w-7xl w-full mx-auto p-4 sm:p-6 lg:p-8 flex flex-col lg:flex-row gap-6">
        
        {/* شاشة إدخال وتعديل البيانات */}
        <div className={`no-print flex-1 lg:max-w-md ${activeTab === "preview" ? "hidden" : "block"}`}>
          
          {activeTab === "step1" && (
            <div className="bg-white rounded-xl shadow-md border border-slate-200 overflow-hidden">
              <div className="bg-slate-800 text-slate-100 p-4 border-b border-amber-500">
                <div className="flex items-center justify-between">
                  <div className="flex items-center gap-2">
                    <User className="text-amber-500" size={18} />
                    <h2 className="font-bold text-sm tracking-wide">الورقة الأولى: بيانات الترخيص والمخول</h2>
                  </div>
                </div>
                <p className="text-xs text-slate-400 mt-1">
                  قم بملء البيانات الأساسية للموظف المخول وسائق المركبة لتفعيل الترخيص.
                </p>
              </div>

              <div className="p-5 space-y-4">
                <div>
                  <label className="block text-xs font-bold text-slate-700 mb-2">نوع الحركة الرسمية للمواد</label>
                  <div className="grid grid-cols-2 gap-3">
                    <button
                      type="button"
                      onClick={() => handlePage1Change("entryExit", "entry")}
                      className={`py-2 px-3 rounded-lg border text-xs font-bold transition-all flex items-center justify-center gap-2 cursor-pointer ${
                        formData.page1.entryExit === "entry"
                          ? "bg-blue-50 border-blue-500 text-blue-700 ring-2 ring-blue-100"
                          : "bg-white border-slate-200 text-slate-600 hover:bg-slate-50"
                      }`}
                    >
                      <span className={`w-3 h-3 rounded-full border flex items-center justify-center ${formData.page1.entryExit === "entry" ? "bg-blue-600 border-transparent" : "border-slate-300"}`}>
                        {formData.page1.entryExit === "entry" && <Check size={10} className="text-white" />}
                      </span>
                      <span>إدخال مواد (استيراد/توريد)</span>
                    </button>
                    
                    <button
                      type="button"
                      onClick={() => handlePage1Change("entryExit", "exit")}
                      className={`py-2 px-3 rounded-lg border text-xs font-bold transition-all flex items-center justify-center gap-2 cursor-pointer ${
                        formData.page1.entryExit === "exit"
                          ? "bg-amber-50 border-amber-500 text-amber-700 ring-2 ring-amber-100"
                          : "bg-white border-slate-200 text-slate-600 hover:bg-slate-50"
                      }`}
                    >
                      <span className={`w-3 h-3 rounded-full border flex items-center justify-center ${formData.page1.entryExit === "exit" ? "bg-amber-600 border-transparent" : "border-slate-300"}`}>
                        {formData.page1.entryExit === "exit" && <Check size={10} className="text-white" />}
                      </span>
                      <span>إخراج مواد (تصدير/إرسال)</span>
                    </button>
                  </div>
                </div>

                <div className="border-t border-slate-100 my-2"></div>

                <div className="space-y-3">
                  <h3 className="text-xs font-bold text-indigo-700">بيانات المسؤول المخول بالاستلام والتسليم</h3>
                  
                  <div>
                    <label className="text-xs font-bold text-slate-600 block mb-1">اسم المسؤول ثلاثي</label>
                    <input
                      type="text"
                      className="w-full text-xs p-2.5 border border-slate-200 rounded-lg outline-none"
                      placeholder="املاء اسم المسؤول المخول الثلاثي"
                      value={formData.page1.authName}
                      onChange={(e) => handlePage1Change("authName", e.target.value)}
                    />
                  </div>

                  <div>
                    <label className="text-xs font-bold text-slate-600 block mb-1">رقم هوية الأحوال للمسؤول</label>
                    <input
                      type="text"
                      className="w-full text-xs p-2.5 border border-slate-200 rounded-lg outline-none"
                      placeholder="مثال: IQ-09184024"
                      value={formData.page1.authId}
                      onChange={(e) => handlePage1Change("authId", e.target.value)}
                    />
                  </div>
                </div>

                <div className="space-y-3 pt-2">
                  <h3 className="text-xs font-bold text-indigo-700">معلومات سائق العجلة المرافقة</h3>

                  <div>
                    <label className="text-xs font-bold text-slate-600 block mb-1">اسم السائق الثلاثي</label>
                    <input
                      type="text"
                      className="w-full text-xs p-2.5 border border-slate-200 rounded-lg outline-none"
                      placeholder="املاء اسم السائق المرافق كامل"
                      value={formData.page1.driverName}
                      onChange={(e) => handlePage1Change("driverName", e.target.value)}
                    />
                  </div>

                  <div>
                    <label className="text-xs font-bold text-slate-600 block mb-1">رقم هوية السائق الوطنية</label>
                    <input
                      type="text"
                      className="w-full text-xs p-2.5 border border-slate-200 rounded-lg outline-none"
                      placeholder="مثال: IQ-23214152"
                      value={formData.page1.driverId}
                      onChange={(e) => handlePage1Change("driverId", e.target.value)}
                    />
                  </div>
                </div>

                <div className="space-y-3 pt-2">
                  <h3 className="text-xs font-bold text-indigo-700">مواصفات المركبة (العجلة)</h3>

                  <div className="grid grid-cols-2 gap-2">
                    <div>
                      <label className="text-xs font-bold text-slate-600 block mb-1">نوع المركبة</label>
                      <select
                        className="w-full text-xs p-2.5 border border-slate-200 rounded-lg outline-none bg-white font-sans"
                        value={formData.page1.vehicleType}
                        onChange={(e) => handlePage1Change("vehicleType", e.target.value)}
                      >
                        <option value="">-- اختر النوع --</option>
                        <option value="شاحنة حمل كيا">شاحنة حمل كيا</option>
                        <option value="سيارة بيك اب">سيارة بيك اب</option>
                        <option value="شاحنة تريلة كبيرة">شاحنة تريلة كبيرة</option>
                        <option value="سيارة صالون خصوصي">سيارة صالون خصوصي</option>
                        <option value="مركبة نقل ستوتة/دراجة">مركبة نقل ستوتة</option>
                        <option value="أخرى / مادة يدوية">أخرى / شحن يدوي</option>
                      </select>
                    </div>

                    <div>
                      <label className="text-xs font-bold text-slate-600 block mb-1">رقم لوحة العجلة</label>
                      <input
                        type="text"
                        className="w-full text-xs p-2.5 border border-slate-200 rounded-lg outline-none"
                        placeholder="مثال: بغداد - 14932 / أ"
                        value={formData.page1.vehicleNumber}
                        onChange={(e) => handlePage1Change("vehicleNumber", e.target.value)}
                      />
                    </div>
                  </div>
                </div>

                <div className="space-y-3 pt-2">
                  <h3 className="text-xs font-bold text-indigo-700">منشأ الشحنة والتاريخ</h3>

                  <div>
                    <label className="text-xs font-bold text-slate-600 block mb-1">منشأ أو وجهة المواد (الدائرة/القسم)</label>
                    <input
                      type="text"
                      className="w-full text-xs p-2.5 border border-slate-200 rounded-lg outline-none"
                      placeholder="ما هي الدائرة مصدر هذه الشحنة؟"
                      value={formData.page1.materialSource}
                      onChange={(e) => handlePage1Change("materialSource", e.target.value)}
                    />
                  </div>

                  <div>
                    <label className="text-xs font-bold text-slate-600 block mb-1">تاريخ الترخيص الفعلي</label>
                    <input
                      type="date"
                      className="w-full text-xs p-2.5 border border-slate-200 rounded-lg outline-none"
                      value={formData.page1.date}
                      onChange={(e) => handlePage1Change("date", e.target.value)}
                    />
                  </div>
                </div>

                <div className="space-y-3 pt-2 bg-slate-50 p-3 rounded-lg border border-slate-150">
                  <h3 className="text-xs font-bold text-slate-850 flex items-center justify-between">
                    <span>تفاصيل الأعداد (الصفحة الأولى)</span>
                    <button
                      type="button"
                      onClick={autoSyncMaterialCount}
                      className="text-[10px] text-blue-600 font-bold hover:underline flex items-center gap-1 cursor-pointer"
                    >
                      <RefreshCw size={10} />
